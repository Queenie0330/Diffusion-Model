WINDOW = 64
STRIDE = 1

class WindowDataset(Dataset):
    def __init__(self, arr, start_idx, end_idx, window=64, stride=1):
        self.arr = arr
        self.window = window
        self.stride = stride
        self.idxs = list(range(start_idx, end_idx - window + 1, stride))
    def __len__(self):
        return len(self.idxs)
    def __getitem__(self, i):
        s = self.idxs[i]
        x = self.arr[s:s+self.window]     # [W, N]
        x = torch.from_numpy(x).T         # -> [N, W]
        return x

train_ds = WindowDataset(rets_z, 0, train_end, window=WINDOW, stride=STRIDE)
val_ds   = WindowDataset(rets_z, train_end - WINDOW, val_end, window=WINDOW, stride=STRIDE)

import math
import torch.nn as nn

def make_beta_schedule(T_steps=200, beta_start=1e-4, beta_end=0.02):
    return torch.linspace(beta_start, beta_end, T_steps)

T_STEPS = 200
betas = make_beta_schedule(T_STEPS).to(device)
alphas = 1.0 - betas
alpha_bars = torch.cumprod(alphas, dim=0)

def q_sample(x0, t, noise):
    a_bar = alpha_bars[t].view(-1, 1, 1)
    return torch.sqrt(a_bar) * x0 + torch.sqrt(1.0 - a_bar) * noise

class TimeEmbedding(nn.Module):
    def __init__(self, dim):
        super().__init__()
        self.dim = dim
        self.mlp = nn.Sequential(
            nn.Linear(dim, dim*4),
            nn.SiLU(),
            nn.Linear(dim*4, dim)
        )
    def forward(self, t):
        half = self.dim // 2
        freqs = torch.exp(-math.log(10000) * torch.arange(0, half, device=t.device).float() / half)
        args = t.float().unsqueeze(1) * freqs.unsqueeze(0)
        emb = torch.cat([torch.sin(args), torch.cos(args)], dim=1)
        if self.dim % 2 == 1:
            emb = torch.cat([emb, torch.zeros_like(emb[:, :1])], dim=1)
        return self.mlp(emb)

class Denoiser1D(nn.Module):
    def __init__(self, n_assets, hidden=128, t_dim=128):
        super().__init__()
        self.t_embed = TimeEmbedding(t_dim)
        self.in_proj = nn.Conv1d(n_assets, hidden, kernel_size=3, padding=1)
        self.block1 = nn.Sequential(
            nn.GroupNorm(8, hidden),
            nn.SiLU(),
            nn.Conv1d(hidden, hidden, kernel_size=3, padding=1),
        )
        self.block2 = nn.Sequential(
            nn.GroupNorm(8, hidden),
            nn.SiLU(),
            nn.Conv1d(hidden, hidden, kernel_size=3, padding=1),
        )
        self.out = nn.Conv1d(hidden, n_assets, kernel_size=3, padding=1)
        self.t_to_hidden = nn.Linear(t_dim, hidden)

    def forward(self, x, t):
        h = self.in_proj(x)
        te = self.t_embed(t)
        h = h + self.t_to_hidden(te).unsqueeze(-1)
        h = h + self.block1(h)
        h = h + self.block2(h)
        return self.out(h)

model = Denoiser1D(n_assets=N).to(device)
opt = torch.optim.AdamW(model.parameters(), lr=2e-4)

def train_one_epoch():
    model.train()
    losses = []
    for x0 in train_loader:
        x0 = x0.to(device)
        B = x0.size(0)
        t = torch.randint(0, T_STEPS, (B,), device=device)
        noise = torch.randn_like(x0)
        xt = q_sample(x0, t, noise)
        pred = model(xt, t)
        loss = (pred - noise).pow(2).mean()

        opt.zero_grad()
        loss.backward()
        opt.step()
        losses.append(loss.item())

@torch.no_grad()
def p_sample_step(xt, t):
    B = xt.size(0)
    t_batch = torch.full((B,), t, device=device, dtype=torch.long)
    eps = model(xt, t_batch)

    beta_t = betas[t]
    alpha_t = alphas[t]
    a_bar_t = alpha_bars[t]
    a_bar_prev = alpha_bars[t-1] if t > 0 else torch.tensor(1.0, device=device)

    x0_pred = (xt - torch.sqrt(1 - a_bar_t) * eps) / torch.sqrt(a_bar_t)

    coef1 = torch.sqrt(a_bar_prev) * beta_t / (1 - a_bar_t)
    coef2 = torch.sqrt(alpha_t) * (1 - a_bar_prev) / (1 - a_bar_t)
    mean = coef1 * x0_pred + coef2 * xt

    if t == 0:
        return mean
    noise = torch.randn_like(xt)
    var = beta_t * (1 - a_bar_prev) / (1 - a_bar_t)
    return mean + torch.sqrt(var) * noise

@torch.no_grad()
def denoise_window(x0_window, start_t=120, K=4):
    x0 = x0_window.unsqueeze(0).to(device)
    outs = []
    for _ in range(K):
        noise = torch.randn_like(x0)
        t0 = torch.tensor([start_t], device=device)
        xt = q_sample(x0, t0, noise)
        for t in reversed(range(0, start_t+1)):
            xt = p_sample_step(xt, t)
        outs.append(xt.squeeze(0).cpu())
    return torch.stack(outs, dim=0).mean(dim=0)
